    import binascii

def _print(promot, msg):
    print('\n'+binascii.hexlify(msg).decode()+'\n')

xor = lambda p, q: bytes(x^y for x,y in zip(p,q))

### >>> b'0000000000'
plaintext_0_suffix = b'0'*10
_print('>>>', plaintext_0_suffix)

###### first respones in hex

_print('>>>', r0)

iv = r0[:16]

text = b'Alice'+b'\x00'*11

### >>>

    import cv2

cap = cv2.VideoCapture('GoodVideo.mp4')

i = 0
while(cap.isOpened()):
    ret, frame = cap.read()
    
    if ret:
        i += 1
        
        name = "img2/frame%s.jpg"%str(i).rjust(10, '0')
        cv2.imwrite(name, frame)
        
        print(f'\rframe={i}', end='')
        #cv2.imshow('Frame',frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break
    else:
        break

cap.release()
cv2.destroyAllWindows()
